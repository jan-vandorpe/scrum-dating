ALLE DATA VAN GEBRUIKER QUERY --------

SELECT a.gebruikerId, a.email, a.geslacht, a.wachtwoord, a.geboorteDatum, a.naam, a.voornaam, a.postcode, a.stad, a.lengte, b.lichaamstype, c.opleidingsNiveau, beroep, d.etnischeAchtergrond, roker, e.oogkleur, aantalKinderen, f.haarkleur, foto, persoonlijkheidsType FROM GEBRUIKER a

INNER JOIN lichaamstypes b
ON a.lichaamsbouwId = b.lichaamstypeId

INNER JOIN opleidingsniveaus c
ON a.hOplNiveauId = c.oplNiveauId

INNER JOIN etnischeachtergronden d
ON a.etnischeAchtergrondId = d.etnischeAchtergrondId

INNER JOIN oogkleuren e
ON a.oogkleurId = e.oogkleurId

INNER JOIN haarkleuren f
on a.haarkleurId = f.haarkleurId

WHERE a.gebruikerId = 1
